# usage:

## Full

```js
import UI from "@rqjs/rqcomponents";

Vue.use(UI);

new Vue({
    el: "#rq-header",
    template: "<RqHeader></RqHeader>"
})
```

## Package


```js
import {RqHeader} from "@rqjs/rqcomponents";

Vue.use(RqHeader);

new Vue({
    el: "#rq-header",
    template: "<RqHeader></RqHeader>"
})
```
